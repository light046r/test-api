Covid19 API

Berfungsi untuk mendapatkan data covid Indonesia. Data diambil dari website www.covid19.go.id

• Get Covid Indonesia

Mendapatkan data covid di Indonesia

Usage :

Example :

[+] NodeJs

const get = require("request");
 get("https://z-rapi.herokuapp.com/api/covid-indo", (err, res, body) => {
  if(!err || res.statusCode == 200){				  
   var result = JSON.parse(res.body);
   console.log(result);
  }
  else {
   throw err;
  }
 });

[+] PHP

<?php
 $c = curl_init();									   
 curl_setopt($c, CURLOPT_URL, "https://z-rapi.herokuapp.com/api/covid-indo");
 curl_setopt($c, CURLOPT_RETURNTRANSFER, 1);
 curl_setopt ( $c , CURLOPT_USERAGENT , 'Mozilla / 5.0 (Linux; Android 5.0.2; Redmi Note 3) AppleWebKit / 537.36 (KHTML, seperti Gecko) Chrome / 78.0.3904.96 Mobile Safari / 537.36' );
 $result = curl_exec($c);
 curl_close($c);
 echo $result;
?>
